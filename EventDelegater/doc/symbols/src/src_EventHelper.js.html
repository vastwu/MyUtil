1 /**
  2  * @fileOverview dom元素事件代理辅助对象
  3  * @author vastwu

  5  */
  6 (function (namespace) {
  7     var NAME = 'EventHelper';
  8     var E = {
  9         FOCUS:'focus',
 10         BLUR:'blur'
 11     };
 12     /**
 13      * @class js代理事件辅助类，用于帮助构建dom树事件代理
 14      * @name EventHelper
 15      * @constructor
 16      * @param {HTMLElement} container dom事件代理顶级元素,所有的事件均由此元素捕获
 17      * @param {Object} [option] 配置项
 18      * @param {String} [option.evtAttr=ek] 触发事件的元素标记属性名,只有携带次属性的元素才会响应事件
 19      */
 20     var EventHelper = function(container, option){
 21         if(!option) option = {};
 22         this._opt = {
 23             'evtAttr':option.evtAttr || 'ek'
 24         };
 25         this._container = container;
 26         this._events = {};
 27         this._eventsCount = {};
 28         this._listeners = {};
 29         //判断 p 是否包含 c, p==c视为包含
 30         var contains = (function(){
 31             return document.body.contains ?
 32                 function(p, c){
 33                     //由窗口外移入移出可能出现p&c不存在的情况
 34                     return (p && c) ? p.contains(c):true;
 35                 }:
 36                 function(p, c){
 37                     return (p && c) ? !!(p.compareDocumentPosition(c) & 16):true;
 38                 };
 39         })();
 40         this._checkers = {
 41             //mousever事件，由自身之外的元素移出,且移入到自身才出发
 42             'mouseover':function(evt, fireElem){
 43                 return !contains(fireElem, evt.fromElement);
 44             },
 45             //mouseout事件，进入到一个自己之外的元素才触发
 46             //fromElement 移出
 47             //toElement 移入
 48             'mouseout':function(evt, fireElem){
 49                 return !contains(fireElem, evt.toElement);
 50             }
 51         };
 52         var c = 0;
 53         this._getRnd = function(){
 54             return 'eID' + c++;
 55         };
 56     };
 57     EventHelper.prototype = {
 58         /**
 59          * @description 绑定dom事件
 60          * @param {HTMLElement} elem 绑定事件的元素
 61          * @param {String} eventType 绑定的事件类型
 62          * @param {Function} eventHandler 事件处理函数
 63          * @return {Function} 最终绑定的事件函数,用于移除事件
 64          */
 65         _bind:(function(){
 66             var e = document.createElement('div');
 67             if ( e.addEventListener ) {
 68                 return function(elem, eventType, eventHandler){
 69                     var cap = (eventType == E.FOCUS || eventType == E.BLUR) ? true : false;
 70                     var _handler = function(e){
 71                         eventHandler.call(elem, e);
 72                     };
 73                     elem.addEventListener( eventType, _handler, cap );
 74                     return _handler;
 75                 };
 76             } else if ( e.attachEvent ) {
 77                 return function(elem, eventType, eventHandler){
 78                     if(eventType == E.FOCUS){
 79                         eventType = 'focusin';
 80                     }
 81                     if(eventType == E.BLUR){
 82                         eventType = 'focusout';
 83                     }
 84                     var _handler = function(e){
 85                         e.target = e.target ? e.target : e.srcElement;
 86                         // e.relatedTarget = e.fromElement;
 87                         eventHandler.call(elem, e);
 88                     };
 89                     elem.attachEvent( "on" + eventType, _handler );
 90                     return _handler;
 91                 };
 92             }
 93         })(),
 94         /**
 95          * @description 移除dom事件
 96          * @param {HTMLElement} elem 绑定事件的元素
 97          * @param {String} eventType 绑定的事件类型
 98          * @param {Function} eventHandler 需要移除的事件函数
 99          */
100         _unbind:(function(){
101             var e = document.createElement('div');
102             if ( e.removeEventListener ) {
103                 return function(elem, eventType, eventHandler){
104                     var cap = (eventType == E.FOCUS || eventType == E.BLUR) ? true : false;
105                     elem.removeEventListener( eventType, eventHandler, cap );
106                 };
107             }else if ( e.detachEvent ) {
108                 return function(elem, eventType, eventHandler){
109                     if(eventType == E.FOCUS){
110                         eventType = 'focusin';
111                     }
112                     if(eventType == E.BLUR){
113                         eventType = 'focusout';
114                     }
115                     elem.detachEvent( "on" + eventType, eventHandler );
116                 };
117             }
118         })(),
119         /**
120          * 触发一个事件
121          * @param  {HTMLElement}  target 触发事件的dom元素
122          * @param  {String}  evtType 出发的事件类型
123          * @param  {event}  evt 浏览器事件对象
124          * @param  {Number}  deepness 冒泡深度,0为首个直接操作元素,渐增
125          * @return {[type]} 事件函数返回值,无限定类型,如果冒泡触发了多个元素事件,只返回最后一次
126          */
127         _fire:function(target, evtType, evt, deepness){
128             var elem = target;
129             var isStop = false;
130             var attr = this._opt.evtAttr;
131             var eventsByType = this._events[evtType];
132             if(!eventsByType || !elem){
133                 return;
134             }
135             var returnValue = true;
136             var ek = elem.getAttribute(attr);
137             if(ek){
138                 var eventsByKey = eventsByType[ek];
139                 if(eventsByKey){
140                     for(var i in eventsByKey){
141                         var eItem = eventsByKey[i];
142                         //有self参数的事件只响应首次事件触发
143                         if(eItem.self === false || deepness === 0){
144                             //执行按事件类型默认的条件检查函数
145                             if(!this._checkers[evtType] || this._checkers[evtType](evt, elem)){
146                                 //对于重复触发的事件，只返回最后一次的返回值
147                                 //dom事件多用于阻止submit提交等操作
148                                returnValue = eItem.handler.call(elem, evt);
149                             }
150                         }
151                         //self参数也有阻止冒泡的功能，具体场景是当其被触发时
152                         if(eItem.isStopPropagation === true || (eItem.self === true && deepness === 0)){
153                             isStop = true;
154                         }
155                         if(eItem.isPreventDefault === true){
156                             evt.returnValue = false;
157                             evt.preventDefault && evt.preventDefault();
158                         }
159                     }
160                 }
161             }
162             if(!isStop && elem && elem !== document.body){
163                 returnValue = this._fire(elem.parentNode, evtType, evt, ++deepness);
164             }
165             return returnValue;
166         },
167         /**
168          * @description 绑定代理事件,在此绑定的事件,不会随着子元素消失而消失,只要存在匹配evtAttr的dom元素,就会触发
169          * @method on
170          * @param {String} listnerCommand  绑定代理事件的命令行, 基础语法为 evtKey:evtType/config1/config2...., 且支持以';'分割的多重事件绑定统一的函数                   
171          *          config支持项
172          *              stopPropagation 阻止继续触发父元素事件，子元素的所有冒泡均会截止至次
173          *              preventDefault  阻止浏览器的默认事件
174          *              self 仅响应直接触发自身的情况，不冒泡，但也不会阻止子元素的冒泡
175          *              
176          * @param  {Function} handler      事件响应函数
177          * @return {String | String[]}     删除该事件依据的参数,若listnerCommand包含多个命令,则会返回这些命令的listener的数组
178          * @example
179          *     var listeners = 
180          *     evtHelper.on('nameLabel:click/stopPropagation;phoneLabel:click', function(e){
181          *     
182          *     });
183          */
184         on:function(listnerCommand, handler){
185             if(listnerCommand.indexOf(';') > -1){
186                 var listeners = [];
187                 var lists = listnerCommand.split(';');
188                 for(var i = 0, n = lists.length; i < n; i++){
189                     listeners.push(this.on(lists[i], handler));
190                 }
191                 return listeners;
192             }
193             //evtKey:evtType/config
194             var command = listnerCommand.match(/([^:]*):([\w]*)\/?(.*)/);
195             if(command.length == 0){
196                 throw new Error('wrong command format, must be evtKey:evtType/config1/config2....');
197             }
198             var evtKey = command[1];
199             var evtType = command[2];
200             var config = command[3];
201             var events = this._events;
202             if(!events[evtType]){
203                 events[evtType] = {};
204                 this._eventsCount[evtType] = 0;
205                 var self = this;
206                 this._listeners[evtType] = this._bind(this._container, evtType, function(evt){
207                     return self._fire(evt.target, evtType, evt, 0);
208                 });
209             }
210             if(!events[evtType][evtKey]){
211                 events[evtType][evtKey] = [];
212             }
213             var eID = this._getRnd();
214             events[evtType][evtKey][eID] = {
215                 'self':config.indexOf('self') > -1,
216                 'isStopPropagation':config.indexOf('stopPropagation') > -1,
217                 'isPreventDefault':config.indexOf('preventDefault') > -1,
218                 'handler':handler
219             };
220             this._eventsCount[evtType]++;
221             return evtKey + ':' + evtType + '/' + eID;
222         },
223         /**
224          * @description 移除事件监听
225          * @param  {String} listnerCommand 移除事件监听的命令，支持以下几种类型
226          *                  on 的返回值，用于精确移除指定的监听函数
227          *                  eventKey/*   移除该eventKey下的所有类型事件监听
228          *                  eventKey/eventType 移除该eventKey下的eventType的事件类型监听
229          */
230         off:function(listnerCommand){
231             //blueText:mouseover
232             //eid
233             var command = listnerCommand.match(/([^:]*):([\w|*]*)\/?(.*)/);
234             var evtKey = command[1];
235             var evtType = command[2];
236             var eID = command[3];
237             //精确删除事件
238             if(eID && this._events[evtType] &&
239                 this._events[evtType][evtKey] &&
240                 this._events[evtType][evtKey][eID]){
241                 delete this._events[evtType][evtKey][eID];
242                 this._eventsCount[evtType]--;
243             }else if(evtType === '*'){
244                 //删除某个key下的所有事件类型
245                 for(var _evtType in this._events){
246                     if(this._events[_evtType].hasOwnProperty(evtKey)){
247                         delete this._events[_evtType][evtKey];
248                         this._eventsCount[_evtType]--;
249                     }
250                 }
251             }else{
252                 //默认情况，删除指定key下的指定事件类型
253                 var eventsByType = this._events[evtType];
254                 if(eventsByType[evtKey]){
255                     delete eventsByType[evtKey];
256                     this._eventsCount[evtType]--;
257                 }
258             }
259             //检查每个事件类型是否还有监听
260             for(var k in this._eventsCount){
261                 if(this._eventsCount[k] <= 0){
262                     this._unbind(this._container, evtType, this._listeners[evtType]);
263                     delete this._events[evtType];
264                 }
265             }
266         },
267         /**
268          * @description 触发事件,不会改变this作用域，也没有冒泡的对应处理，仅仅会执行匹配的函数
269          * @param  {[type]} evtKey 事件key
270          * @param  {[type]} evtType 事件类型
271          * @param  {[type]} args 需要传入的参数,数组形式,且真实的参数首个为null,填充event的位置,使得被执行函数可以知道是来源于trigger
272          */
273         trigger:function(evtKey, evtType, args){
274             var eventsByType = this._events[evtType] || {};
275             var eventsByKey = eventsByType[evtKey];
276             if(!eventsByKey){
277                 return;
278             }
279             var returnValue = true;
280             args = args || [];
281             //插入null以替代event，让被trigger的函数可以明确知道自己是如何被调用的
282             args.unshift(null);
283             for(var i in eventsByKey){
284                 //对于重复触发的事件，只返回最后一次的返回值
285                 //dom事件多用于阻止submit提交等操作
286                 returnValue = eventsByKey[i].handler.apply(null, args);
287             }
288             return returnValue;
289         },
290         /**
291          * 清除所有事件和监听
292          */
293         clear:function(){
294             for(var evtType in this._listeners){
295                 this._unbind(this._container, evtType, this._listeners[evtType]);
296                 delete this._events[evtType];
297                 delete this._listeners[evtType];
298             }
299         }
300     };

313         /**
314          * 克隆自身,禁止复写

320             return new this.constructor(this._option);
321         },
322         /**
323          * 复写接口,会在this上保存一份最后一次覆盖前的函数
324          * @param  {[type]} methodName [description]
325          * @param  {[type]} handler    复写函数
326          * @return {[type]}            [description]

328         override:function(methodName, handler){
329             if(this[methodName]){
330                 this['overrided_' + methodName] = this[methodName];

390          * @param  {Object} infowindow infowindow组件对象
391          */
392         bindEvent:function(evtHelper, info, overlay, infowindow){
393             this.$super.bindEvent(evtHelper, info, overlay, infowindow);
394         }
